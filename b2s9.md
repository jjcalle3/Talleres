import rx.lang.scala.Observable
import scala.concurrent.duration._
import scala.util.Random

object TallerInd1_Bim2 extends App {

  val sensor1 = Observable.interval(1.second).map(_ => Random.between(20, 30))
  val sensor2 = Observable.interval(1.5.second).map(_ => Random.between(30, 40))


  val datosComb = sensor1.zip(sensor2)


  datosComb.take(10).subscribe(
    onNext = { case (temp1, temp2) =>
      println(s"Sensor 1: $temp1¬∞C, Sensor2: $temp2¬∞C")

      // C√°lculo de la diferencia de temperatura y alerta si es mayor a 5¬∞C
      val diff = math.abs(temp1 - temp2)
      if (diff > 5) {
        println(f"üö® ALERTA! Alta temperatura detectada: $diff%.2f¬∞C")
      }
    },
    onError = e => println(s"Error: ${e.getMessage}"),
    onCompleted = () => println("Flujo completado")
  )


  Thread.sleep(15000)
}
